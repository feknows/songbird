## Supabase setup

1. Create a Supabase project.
2. Copy `.env.example` to `.env` locally (for `migrate-supabase.js`).
3. **Critical**: In `public/index.html:609-610`, replace the placeholder credentials with real Supabase URL and anon key.
4. Run the SQL below in the SQL Editor (creates schema + RLS policies):
   ```sql
   -- SongBird - Supabase Schema
   CREATE TABLE IF NOT EXISTS usuarios (
     id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
     nome TEXT NOT NULL UNIQUE,
     senha TEXT NOT NULL,
     permissao TEXT NOT NULL DEFAULT 'user' CHECK (permissao IN ('admin', 'user')),
     criado_em TIMESTAMPTZ DEFAULT NOW()
   );
   CREATE TABLE IF NOT EXISTS produtos (
     id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
     nome TEXT NOT NULL,
     tipo TEXT NOT NULL,
     ativo INTEGER DEFAULT 1 CHECK (ativo IN (0, 1)),
     ordem INTEGER DEFAULT 0
   );
   CREATE TABLE IF NOT EXISTS faixas (
     id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
     produto_id BIGINT NOT NULL REFERENCES produtos(id) ON DELETE CASCADE,
     descricao TEXT NOT NULL,
     valor_base NUMERIC(10,2) NOT NULL DEFAULT 0,
     valor_modulo NUMERIC(10,2) NOT NULL DEFAULT 0,
     ordem INTEGER DEFAULT 0
   );
   CREATE INDEX IF NOT EXISTS idx_faixas_produto_id ON faixas(produto_id);
   CREATE TABLE IF NOT EXISTS modulos (
     id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
     produto_id BIGINT NOT NULL REFERENCES produtos(id) ON DELETE CASCADE,
     nome TEXT NOT NULL,
     ativo INTEGER DEFAULT 1 CHECK (ativo IN (0, 1)),
     ordem INTEGER DEFAULT 0
   );
    CREATE INDEX IF NOT EXISTS idx_modulos_produto_id ON modulos(produto_id);
   CREATE TABLE IF NOT EXISTS modulo_precos (
     id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
     modulo_id BIGINT NOT NULL REFERENCES modulos(id) ON DELETE CASCADE,
     faixa_id BIGINT NOT NULL REFERENCES faixas(id) ON DELETE CASCADE,
     valor NUMERIC(10,2) NOT NULL DEFAULT 0,
     UNIQUE(modulo_id, faixa_id)
   );
   CREATE INDEX IF NOT EXISTS idx_modulo_precos_modulo_id ON modulo_precos(modulo_id);
   CREATE INDEX IF NOT EXISTS idx_modulo_precos_faixa_id ON modulo_precos(faixa_id);
   CREATE TABLE IF NOT EXISTS leads (
     id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
     nome TEXT NOT NULL,
     telefone TEXT DEFAULT '',
     whatsapp TEXT DEFAULT '',
     site TEXT DEFAULT '',
     endereco TEXT DEFAULT '',
     avaliacao NUMERIC(3,1),
     redes_sociais TEXT DEFAULT '',
     cidade TEXT NOT NULL,
     termo TEXT NOT NULL,
     criado_em TIMESTAMPTZ DEFAULT NOW()
   );
   -- RLS policies: SELECT livre, escrita apenas autenticado
   ALTER TABLE usuarios ENABLE ROW LEVEL SECURITY;
   ALTER TABLE produtos ENABLE ROW LEVEL SECURITY;
   ALTER TABLE faixas ENABLE ROW LEVEL SECURITY;
    ALTER TABLE modulos ENABLE ROW LEVEL SECURITY;
   ALTER TABLE modulo_precos ENABLE ROW LEVEL SECURITY;
   ALTER TABLE leads ENABLE ROW LEVEL SECURITY;
   DROP POLICY IF EXISTS "usuarios_select" ON usuarios;
   DROP POLICY IF EXISTS "usuarios_update" ON usuarios;
   DROP POLICY IF EXISTS "usuarios_delete" ON usuarios;
   CREATE POLICY "usuarios_select" ON usuarios FOR SELECT USING (true);
   CREATE POLICY "usuarios_update" ON usuarios FOR UPDATE
     USING (auth.uid()::text = id::text)
     WITH CHECK (auth.uid()::text = id::text);
   CREATE POLICY "usuarios_delete" ON usuarios FOR DELETE
     USING (auth.role() = 'service_role');
   DROP POLICY IF EXISTS "produtos_select" ON produtos;
   DROP POLICY IF EXISTS "produtos_insert" ON produtos;
   DROP POLICY IF EXISTS "produtos_update" ON produtos;
   DROP POLICY IF EXISTS "produtos_delete" ON produtos;
   CREATE POLICY "produtos_select" ON produtos FOR SELECT USING (true);
   CREATE POLICY "produtos_insert" ON produtos FOR INSERT
     WITH CHECK (auth.role() = 'authenticated' OR auth.role() = 'service_role');
   CREATE POLICY "produtos_update" ON produtos FOR UPDATE
     USING (auth.role() = 'authenticated' OR auth.role() = 'service_role')
     WITH CHECK (auth.role() = 'authenticated' OR auth.role() = 'service_role');
   CREATE POLICY "produtos_delete" ON produtos FOR DELETE
     USING (auth.role() = 'authenticated' OR auth.role() = 'service_role');
   -- (faixas, modulos, leads seguem o mesmo padrão — veja supabase-schema.sql)
   ```
5. Create an admin user via SQL Editor (generate SHA-256 hash with `node -e "console.log(require('crypto').createHash('sha256').update('SUA_SENHA').digest('hex'))"`):
   ```sql
   INSERT INTO usuarios (nome, senha, permissao)
   VALUES ('seu_usuario', 'SEU_HASH_AQUI', 'admin');
   ```

## Commands

| Command | Purpose |
|---------|---------|
| `node migrate-supabase.js` | Migrate from legacy SQLite `dados.db` to Supabase. Requires `npm install @supabase/supabase-js sql.js`. Uses `SUPABASE_SERVICE_ROLE_KEY` do `.env`. |

## Password

- SHA-256, **not bcrypt**. Both client-side login (`public/index.html:613-616`) and server-side migration use it.
- `migrate-supabase.js` reads `dados.db` (SQLite), SHA-256 hashes passwords, batch-upserts to Supabase using **service_role key** (contorna RLS), then renames `dados.db` → `dados.db.bak`.
- O `.env` deve conter `SUPABASE_URL` e `SUPABASE_SERVICE_ROLE_KEY` (não a anon key).
